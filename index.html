const input = document.getElementById("inputImage"); const image = document.getElementById("image"); const result = document.getElementById("result"); let cropper; input.addEventListener("change", function(e) { const file = e.target.files[0]; if (!file) return; image.src = URL.createObjectURL(file); image.onload = function() { if (cropper) { cropper.destroy(); } cropper = new Cropper(image, { viewMode: 1, autoCropArea: 1, movable: true, zoomable: true, scalable: true, rotatable: true, responsive: true, background: false }); }; }); // Crop image with fixed pixel size document.getElementById("cropBtn").onclick = function() { if (!cropper) return; const canvas = cropper.getCroppedCanvas({ width: 800, // Output width in pixels height: 800, // Output height in pixels imageSmoothingEnabled: true, imageSmoothingQuality: "high" }); result.src = canvas.toDataURL("image/png"); }; // Download cropped image document.getElementById("downloadBtn").onclick = function() { if (!cropper) return; const canvas = cropper.getCroppedCanvas({ width: 800, height: 800, imageSmoothingEnabled: true, imageSmoothingQuality: "high" }); canvas.toBlob(function(blob) { const link = document.createElement("a"); link.download = "cropped-image-800x800.png"; link.href = URL.createObjectURL(blob); link.click(); }, "image/png"); };
